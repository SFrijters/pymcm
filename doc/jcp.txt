function jcp(p,callback,loadingDivID)
{
    var r = getRequester();
    var url = "/iajax.php";
    p="args="+p;

    if(loadingDivID)
    {
        var loadingDiv = document.getElementById(loadingDivID);
        var ajaxLoader = document.getElementById("ajaxLoader").cloneNode(false);
        var oldStylePos = getStyle(loadingDivID,"position");

        loadingDiv.appendChild(ajaxLoader);
        if( oldStylePos != "absolute" )
        {
            loadingDiv.style.position = "relative";
        }
        ajaxLoader.style.display = "block";
    }

    r.onreadystatechange=function()
    {
        if(r.readyState==4 && r.status==200)
        {
            if(loadingDivID)
            {
                loadingDiv.style.position = oldStylePos;
                loadingDiv.removeChild(ajaxLoader);
            }

            window[callback](r.responseXML);
        }
    }

    r.open("POST",url,true);
    r.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
    r.setRequestHeader("Content-length", p.length);
    r.setRequestHeader("Connection", "close");
    r.send(p);
}
